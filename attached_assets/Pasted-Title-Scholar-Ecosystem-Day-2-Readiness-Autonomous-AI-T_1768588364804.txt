import numpy as np

def verify_explanation_stability(model, data):
    """Ensures model fairness explanations are stable under noise."""
    base_shap = shap.Explainer(model)(data).values
    noise = np.random.normal(0, 0.01, data.shape)
    new_shap = shap.Explainer(model)(data + noise).values
    
    diff = np.abs(base_shap - new_shap).mean()
    if diff > 0.05:
        print(f"FAIL: Unstable explanations (Diff: {diff})")
        return False
    return True
